==========  Task 2: Simple Addition  ==========
START
BEGIN
	DISPLAY "Enter 1st number: "
	READ NUMBER1
	DISPLAY "Enter 2nd number: "
	READ NUMBER2
	SET SUM TO NUMBER1 + NUMBER2
	DISPLAY "The sum of the two numbers is: " + SUM
END
STOP
==========  Task 3: Circle Perimeter Calculator  ==========
START
BEGIN
	DISPLAY "Enter radius of the circle(cm): "
	READ RADIUS
	SET CIRCLE_PERIMETER TO 2*3.14*RADIUS
	DISPLAY "The perimeter or the circle is: " + CIRCLE_PERIMETER
END
STOP
==========  Task 4: Days to Hours Converter  ==========
START
BEGIN
	DISPLAY "Enter number of days: "
	READ DAYS
	SET HOURS TO DAYS*24
	DISPLAY "Number of hours is: " + HOURS
END
STOP

START
BEGIN

	END IF
END
STOP
==========  Task 6: Sign of a Number  ==========
START
BEGIN
	DISPLAY "Enter the number: "
	READ Number
	IF Number > 0 THEN
		DISPLAY "Positive Number!"
	ELSE
		DISPLAY "Negative Number!"
	END IF
END
STOP
