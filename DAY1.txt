==========  Task 23: Average of N Inputs   ==========
START
BEGIN
	READ N
	SET SUM TO 0
	REPEAT WHILE N GREATER THAN 0
		READ NUMBER
		SET SUM TO SUM + NUMBER
		SET N TO N-1
	END REPEAT
	
	SET AVERAGE TO SUM / Negative
	DISPLAY "The average is: " +  AVERAGE
	
END
STOP
==========  Task 24: High-Low Guessing Core  ==========
START
BEGIN
	SET SECRET TO 42
	SET GUESS TO 0
	REPEAT WHILE GUESS NOT EQUALS SECRET
		READ GUESS
		IF GUESS EQUALS SECRET THEN
			DISPLAY "YOU GUESSED RIGHT"
		ELSE
			IF GUESS GREATER THAN SECRET THEN
				DISPLAY "YOUR GUESS IS GREATER THAN SECRET NUMBER"
			ELSE
				DISPLAY "YOUR GUESS IS LESS THAN SECRET NUMBER"
			END IF
		END IF
	END REPEAT
	
	SET AVERAGE TO SUM / Negative
	DISPLAY "The average is: " +  AVERAGE
	
END
STOP
