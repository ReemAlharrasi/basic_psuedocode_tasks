==========  Task 3: Circle Perimeter Calculator  ==========
START
BEGIN

	SET CIRCLE_PERIMETER TO 2*3.14*RADIUS
	DISPLAY "The perimeter or the circle is: " + CIRCLE_PERIMETER
END
STOP

==========  Task 5: Voting Eligibility Check  ==========
START
BEGIN
	DISPLAY "Enter your age: "
	READ AGE
	
	IF AGE GREATER THAN OR EQUAL TO 18 THEN
		DISPLAY "Allowed to vote!"
	ELSE
		DISPLAY "Not allowed to vote!"
	END IF
END
STOP
==========  Task 6: Sign of a Number  ==========
START
BEGIN
	DISPLAY "Enter the number: "
	READ Number
	
	IF Number GREATER THAN 0 THEN
		DISPLAY "Positive Number!"
	ELSE
		IF Number LESS THAN 0 THEN
			DISPLAY "Negative Number!"
		ELSE
			DISPLAY "ZerO!"
		END IF
	END IF
END
STOP

==========  Task 8: Profit or Loss Tracker  ==========
START
BEGIN
